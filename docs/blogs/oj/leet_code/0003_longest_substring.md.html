    First Idea

    one scaning

    /**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function (s) {
  let result = 0;

  let cache = {};
  let lengthCount = 0;
  let start = 0;

  for (let i = 0; i < s.length; i++) {
    const ch = s.charAt(i);
    const index = cache[ch];

    if (index != undefined) {
      for (let j = start; j <= index; j++) {
        delete cache[s.charAt(j)];
      }

      lengthCount = lengthCount - (index - start + 1);
      start = index + 1;
    }

    lengthCount++;
    cache[ch] = i;
    result = result < lengthCount ? lengthCount : result;
  }

  return result;
};
    • Time Complexity: O(n)
    • Space Complexity: O(n)

    Sliding Window

    My first solution for this problem is similar to Sliding Window, check doc for details.

    /**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function (s) {
  let max = 0;
  let cachedKeys = {};

  let start = 0; // the sliding window
  let end = 0;

  for (; end < s.length; end++) {
    const cachedIndex = cachedKeys[s[end]];

    if (cachedIndex !== undefined) {
      const len = end - start;
      max = max < len ? len : max;

      for (let j = start; j <= cachedIndex; j++) {
        delete cachedKeys[s.charAt(j)];
      }

      start = cachedIndex + 1;
    }

    cachedKeys[s.charAt(end)] = end;
  }

  return max < end - start ? end - start : max;
};
    • Time Complexity: O(n)
    • Space Complexity: O(n)
