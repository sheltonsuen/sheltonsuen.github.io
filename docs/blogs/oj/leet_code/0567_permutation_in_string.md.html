/**
 * @param {string} s1
 * @param {string} s2
 * @return {boolean}
 */
var checkInclusion = function (s1, s2) {
  let map = new Map()
  const reset = () => map.forEach(({ count, cur }, k) => map.set(k, { count, cur: 0 }))

  for (let i = 0; i < s1.length; i++) {
    const ch = s1.charAt(i)
    map.has(ch) ? map.set(ch, { count: map.get(ch).count + 1, cur: 0 }) : map.set(ch, { count: 1, cur: 0 })
  }

  for (let i = 0; i <= s2.length - s1.length; i++) {
    reset()
    for (let j = 0; j < s1.length; j++) {
      const ch = s2.charAt(j + i)
      if (!map.has(ch)) {
        break
      }

      const { count, cur } = map.get(ch)
      if (cur >= count) {
        break
      }

      map.set(ch, { count, cur: cur + 1 })

      if (j === s1.length - 1) {
        return true
      }
    }
  }

  return false
}

-- Time Complexiy: O(n^2) -- Space Complexiy: O(n)
