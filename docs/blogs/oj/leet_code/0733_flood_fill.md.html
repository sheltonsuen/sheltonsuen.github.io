BFS

const DIRS = [
  [0, 1],
  [1, 0],
  [0, -1],
  [-1, 0],
]

/**
 * @param {number[][]} image
 * @param {number} sr
 * @param {number} sc
 * @param {number} newColor
 * @return {number[][]}
 */
var floodFill = function (image, sr, sc, newColor) {
  let queue = [[sr, sc]]
  let visited = Array(image.length)
    .fill()
    .map(() => [])
  const color = image[sr][sc]
  visited[sr][sc] = true
  image[sr][sc] = newColor

  while (queue.length) {
    const [x, y] = queue.shift()

    DIRS.forEach(([_x, _y]) => {
      const [newX, newY] = [x + _x, y + _y]

      if (image[newX]?.[newY] === color && !visited[newX]?.[newY]) {
        image[newX][newY] = newColor
        visited[newX][newY] = true
        queue.push([newX, newY])
      }
    })
  }

  return image
}
