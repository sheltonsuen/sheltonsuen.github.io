/**
 * @param {character[]} s
 * @return {void} Do not return anything, modify s in-place instead.
 */
var reverseString = function (s) {
  let low = 0
  let high = s.length - 1

  while (low < high) {
    const tmp = s[low]
    s[low++] = s[high]
    s[high--] = tmp
  }
}

-- Time Complexity: O(n) -- Space Complexity: O(1)
