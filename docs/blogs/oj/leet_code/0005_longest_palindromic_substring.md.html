First Idea

/**
 * @param {string} s
 * @return {string}
 */
var longestPalindrome = function (s) {
  let result = []

  for (let i = 0; i < s.length; i++) {
    let j = i
    while ((j = s.indexOf(s.charAt(i), j + 1)) > 0) {
      let curR = j

      if (j <= i) {
        continue
      }

      let curL = i

      while (curL++ < curR--) {
        if (s.charAt(curL) !== s.charAt(curR)) {
          break
        }
      }

      if (curL >= curR) {
        result.push(s.substring(i, j + 1))
      }
    }
  }

  return result.reduce(
    (maxStr, cur) => (maxStr.length < cur.length ? cur : maxStr),
    s.charAt(0)
  )
}
  • Time Complexity: O(n^3)
  • Space Complexity: O(n * n)

Refined Version

/**
 * @param {string} s
 * @return {string}
 */
var longestPalindrome = function (s) {
  if (s.length < 2) {
    return s[0]
  }

  let start = 0
  let end = 0

  function expendForPalindromic(i, j) {
    while (i >= 0 && j < s.length && s.charAt(i) === s.charAt(j)) {
      i--
      j++
    }

    if (end - start < j - i - 1) {
      start = i + 1
      end = j
    }
  }

  for (let i = 0; i < s.length; i++) {
    expendForPalindromic(i, i)
    expendForPalindromic(i, i + 1)
  }

  return s.substring(start, end)
}
  • Time Complexity: O(n^2)
  • Space Complexity: O(1)
