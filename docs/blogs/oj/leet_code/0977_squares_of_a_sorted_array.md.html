    First Idea

    /**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortedSquares = function (nums) {
  let right = 0
  while (nums[right] < 0) {
    right++
  }

  for (let i = 0; i < nums.length; i++) {
    nums[i] = nums[i] ** 2
  }

  let result = Array(nums.length)
  let cur = 0
  let left = right - 1

  while (left >= 0 && right < nums.length) {
    result[cur++] = nums[left] < nums[right] ? nums[left--] : nums[right++]
  }

  while (left >= 0) {
    result[cur++] = nums[left--]
  }

  while (right < nums.length) {
    result[cur++] = nums[right++]
  }

  return result
}

    -- Time Complexity: O(n) -- Space Complexity: O(1)
