/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function (nums) {
  if (nums.length === 1) {
    return [nums]
  }

  let result = []
  for (let i = 0; i < nums.length; i++) {
    result = [
      ...result,
      ...permute([...nums.slice(0, i), ...nums.slice(i + 1)]).map((v) => [
        nums[i],
        ...v,
      ]),
    ]
  }

  return result
}
