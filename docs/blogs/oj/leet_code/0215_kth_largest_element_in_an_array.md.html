    First idea

    /**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var findKthLargest = function (nums, k) {
  // should not use the sort method directly
  // since by default, sort will convert the value to string to compare between each other
  // refer to [compareFunction](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array/sort#description)
  return nums.sort((a, b) => a - b)[nums.length - k]
}
    • Time Complexity: O(n*logn) if using quick sort
    • Space Complexity: O(1)

    Priority Queue

    /**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var findKthLargest = function (nums, k) {
  const priorityQueue = []

  for (const num of nums) {
    const last = priorityQueue[k - 1]
    if (last === undefined || num > last) {
      last === undefined
        ? priorityQueue.push(num)
        : (priorityQueue[k - 1] = num)

      priorityQueue.sort((a, b) => b - a)
    }
  }

  return priorityQueue[k - 1]
}
