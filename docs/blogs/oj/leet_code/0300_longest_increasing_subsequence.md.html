/**
 * @param {number[]} nums
 * @return {number}
 */
var lengthOfLIS = function (nums) {
  let maxTails = []
  let max = 0

  for (let i = 0; i < nums.length; i++) {
    let low = 0
    let high = max

    while (low < high) {
      const middle = Math.floor((low + hight) / 2)

      if (maxTails[middle] < nums[i]) {
        low = middle + 1
      } else {
        high = middle
      }
    }
  }

  return max
}
