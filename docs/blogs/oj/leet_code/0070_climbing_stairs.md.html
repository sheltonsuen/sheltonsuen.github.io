Recursive Version

/**
 * @param {number} n
 * @return {number}
 */
var climbStairs = function (n) {
  if (n < 3) {
    return n
  }

  return climbStairs(n - 1) + climbStairs(n - 2)
}

/**
 * @param {number} n
 * @return {number}
 */
var climbStairs = function (n) {
  if (n < 3) {
    return n
  }

  let [prev, cur] = [1, 2]

  for (let i = 3; i <= n; i++) {
    let next = prev + cur
    prev = cur
    cur = next
  }

  return cur
}

-- Time Complexity: O(n) -- Space Complexity: O(1)
