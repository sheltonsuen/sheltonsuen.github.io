    First Idea

    function indexOfDuplicated(str) {
  let prev = 0
  for (let i = 1; i < str.length; i++) {
    if (str.charAt(prev) === str.charAt(i)) {
      return prev
    }

    prev = i
  }

  return -1
}

/**
 * @param {string} s
 * @return {string}
 */
var removeDuplicates = function (s) {
  let cur = s
  let i = -1

  while ((i = indexOfDuplicated(cur)) >= 0) {
    cur = cur.replace(`${cur.charAt(i)}${cur.charAt(i)}`, '')
  }

  return cur
}

    Refined Version

    function findDuplicatedChars(start, s) {
  let prev = start
  for (let i = start + 1; i < s.length; i++) {
    if (s.charAt(prev) === s.charAt(i)) {
      return prev
    }

    prev = i
  }

  return -1
}

function findRemovalRanges(start, s) {
  let left = start
  let right = start + 1

  while (left >= 0 && right < s.length && s.charAt(left) === s.charAt(right)) {
    left--
    right++
  }

  return [left, right]
}

/**
 * @param {string} s
 * @return {string}
 */
var removeDuplicates = function (s) {
  let removals = []
  let index = 0

  while ((index = findDuplicatedChars(index, s)) >= 0) {
    removals.push(findRemovalRanges(index, s))
  }

  let result = []

  for (let i = 0; i < s.length; i++) {
    if (!removals.some(([s, e]) => i >= s && i <= e)) {
      result.push(s.charAt(i))
    }
  }

  return result.join('')
}
